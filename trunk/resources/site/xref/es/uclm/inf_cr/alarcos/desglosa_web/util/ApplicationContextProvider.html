7   /**

12   * Wrapper to always return a reference to the Spring Application Context from
13   * within non-Spring enabled beans. Unlike Spring MVC's
14   * WebApplicationContextUtils we do not need a reference to the Servlet context
15   * for this. All we need is for this bean to be initialized during application
16   * startup.
17   */
18  public class ApplicationContextProvider implements ApplicationContextAware {
19  
20      private static ApplicationContext context;
21  
22      /**
23       * This method is called from within the ApplicationContext once it is done
24       * starting up, it will stick a reference to itself into this bean.
25       * 
26       * @param context
27       *            a reference to the ApplicationContext.
28       */
29      public void setApplicationContext(ApplicationContext ctx)
30              throws BeansException {
31          context = ctx;
32      }
33  
34      /**
35       * This is about the same as context.getBean("beanName"), except it has its
36       * own static handle to the Spring context, so calling this method
37       * statically will give access to the beans by name in the Spring
38       * application context. As in the context.getBean("beanName") call, the
39       * caller must cast to the appropriate target class. If the bean does not
40       * exist, then a Runtime error will be thrown.
41       * 
42       * @param beanName
43       *            the name of the bean to get.
44       * @return an Object reference to the named bean.
45       */
46      public static Object getBean(String beanName) {
47          return context.getBean(beanName);
48      }
49  }
