33      public Map<String, String> getProfiles(String entity) throws JAXBException, IOException, InstantiationException, IllegalAccessException {
34          Map<String, String> profileNames = new HashMap<String, String>();
35          // Create a filename filter
36          FilenameFilter filter = new ProfileFilter(entity);
37          // Get profile folder full path
38          String fullFileName = Utilities.getRealPathToWebApplicationContext(ProfileDAO.PROFILE_FOLDER);
39          File directory = new File(fullFileName);
40          // Get all filenames in directory that applies filename filter
41          // restrictions
42          String[] children = directory.list(filter);
43          // Unmarshal metaclasses
44          for (String fileName : children) {
45              Metaclass mc = XMLAgent.unmarshal(fullFileName + "\\" + fileName, Metaclass.class);
46              // Check that entity is ok
47              String[] entityParts = mc.getEntityName().split("\\.");
48              if (WordUtils.uncapitalize(entityParts[entityParts.length - 1]).equals(entity))
49                  profileNames.put(fileName, mc.getDescription());
50          }
51          return profileNames;
52      }

54      public List<Profile> getProfiles() throws JAXBException, IOException, InstantiationException, IllegalAccessException {
55          List<Profile> profiles = new ArrayList<Profile>();

58          File directory = new File(fullFileName);
59          // Get all filenames in directory that applies filename filter

61          String[] children = directory.list();
62          // Unmarshal metaclasses
63          for (String fileName : children) {
64              Metaclass mc = XMLAgent.unmarshal(fullFileName + "\\" + fileName, Metaclass.class);
65              profiles.add(new Profile(fileName, mc));

80          // Create XML from metaclass and place it in server
