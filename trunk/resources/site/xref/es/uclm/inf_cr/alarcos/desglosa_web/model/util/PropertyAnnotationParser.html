12

14      }

17              throws Exception {
18          return parse(Class.forName(clazz));

21      public static final List<PropertyWrapper> parse(final Class<?> clazz)
22              throws Exception {
23          List<PropertyWrapper> properties = new ArrayList<PropertyWrapper>();
24          List<Field> fields = Arrays.asList(clazz.getDeclaredFields());
25  
26          for (Field field : fields) {
27              if (field.isAnnotationPresent(Property.class)) {
28                  Property property = field.getAnnotation(Property.class);
29                  String name = property.name();
30                  if (name.equals("")) {
31                      name = field.getName();
32                  }
33                  String type = property.type();
34                  if (type.equals("")) {
35                      type = field.getType().getName();
36                      if (type.equals("java.lang.Boolean")) {
37                          type = "boolean";
38                      } else if (type.equals("java.lang.Integer")) {
39                          type = "int";
40                      } else if (type.equals("java.lang.Float")) {
41                          type = "float";
42                      } else if (type.equals("java.lang.String")) {
43                          type = "string";
44                      }
45                  }
46                  // if it has embedded properties, parse its class type by
47                  // recursive call
48                  if (property.embedded()) {
49                      Class<?> c = Class.forName(type);
50                      List<PropertyWrapper> embeddedProperties = parse(c);
51                      // concat parent attribute name for later manipulation
52                      for (PropertyWrapper pw : embeddedProperties) {
53                          pw.setName(name + "_" + pw.getName());
54                      }
55                      // add all returned properties, not the property itself
56                      properties.addAll(embeddedProperties);
57                  } else {
58                      String description = property.description();
59                      // add it
60                      properties
61                              .add(new PropertyWrapper(name, type, description));
62                  }
63              }
64          }
65          return properties;
66      }
