222     private static Map<String, List<?>> groupEntitiesBy(Class<?> clazz, List<?> entities, String groupBy, int groupById) throws EntityNotSupportedException, GroupByOperationNotSupportedException, CompanyNotFoundException, MarketNotFoundException, ProjectNotFoundException, FactoryNotFoundException {
223         String className = clazz.getSimpleName();
224         Map<String, List<?>> neighborhoods = new HashMap<String, List<?>>();
225         if (className.equals("Company")) {
226             neighborhoods = new HashMap<String, List<?>>();
227             neighborhoods.put("", entities);
228         } else if (className.equals("Factory")) {
229             neighborhoods = groupFactoriesBy(entities, groupBy, groupById);
230         } else if (className.equals("Project")) {
231             neighborhoods = groupProjectsBy(entities, groupBy, groupById);
232         } else if (className.equals("Subproject")) {
233             neighborhoods = groupSubprojectsBy(entities, groupBy, groupById);
234         } else {
235             throw new EntityNotSupportedException();
236         }
237         return neighborhoods;
238     }

240     private static Map<String, List<?>> groupFactoriesBy(List<?> factories, String groupBy, int groupById) throws GroupByOperationNotSupportedException, CompanyNotFoundException, MarketNotFoundException, ProjectNotFoundException {
241         Map<String, List<?>> neighborhoods = new HashMap<String, List<?>>();
242         if (groupBy.equals(GROUP_BY_COMPANY)) {
243             List<Company> availableCompanies;
244             if (groupById == 0) {
245                 availableCompanies = CompanyManager.getAllCompanies();
246             } else {
247                 availableCompanies = new ArrayList<Company>();
248                 availableCompanies.add(CompanyManager.getCompany(groupById));
249             }
250             for (Company comp : availableCompanies) {
251                 List<Factory> flats = new ArrayList<Factory>();
252                 for (Object f : factories) {
253                     if (((Factory) f).getCompany().getId() == comp.getId()) {
254                         flats.add((Factory) f);
255                     }
256                 }
257                 if (flats.size() > 0)
258                     neighborhoods.put(comp.getName(), flats);
259             }
260         } else if (groupBy.equals(GROUP_BY_MARKET)) {
261             List<Market> availableMarkets;
262             if (groupById == 0) {
263                 availableMarkets = MarketManager.getAllMarkets();
264             } else {
265                 availableMarkets = new ArrayList<Market>();
266                 availableMarkets.add(MarketManager.getMarket(groupById));
267             }
268             for (Market m : availableMarkets) {
269                 List<Factory> flats = new ArrayList<Factory>();
270                 for (Object f : factories) {
271                     Market market = ((Factory) f).getMostRepresentativeMarket();
272                     if (market != null && market.equals(m)) {
273                         flats.add((Factory) f);
274                     }
275                 }
276                 if (flats.size() > 0) {
277                     neighborhoods.put(m.getName(), flats);
278                 }
279             }
280             // Hago una segunda pasada para crear un barrio sin nombre en caso de que haya factorias sin mercado de especializacion
281             List<Factory> orphanFactories = new ArrayList<Factory>();
282             for (Object f : factories) {
283                 Market market = ((Factory) f).getMostRepresentativeMarket();
284                 if (market == null) {
285                     orphanFactories.add((Factory) f);
286                 }
287             }
288             if (orphanFactories.size() > 0) {
289                 neighborhoods.put("", orphanFactories);
290             }
291         } else if (groupBy.equals(GROUP_BY_PROJECT)) {
292             List<Project> availableProjects;
293             if (groupById == 0) {
294                 availableProjects = ProjectManager.getAllProjects();
295             } else {
296                 availableProjects = new ArrayList<Project>();
297                 availableProjects.add(ProjectManager.getProject(groupById));
298             }
299             for (Project p : availableProjects) {
300                 int factoryIndex = 0;
301                 List<Factory> flats = new ArrayList<Factory>();
302                 for (Object f : factories) {
303                     boolean found = false;
304                     Iterator<?> it = p.getSubprojects().iterator();
305                     while (it.hasNext() && !found) {
306                         Subproject spaux = (Subproject) it.next();
307                         if (spaux.getFactory().getId() == ((Factory) f).getId()) {
308                             flats.add((Factory) f);
309                             found = true;
310                         }
311                     }
312                     factoryIndex++;
313                 }
314                 if (flats.size() > 0)
315                     neighborhoods.put(p.getName(), flats);
316             }
317         } else if (groupBy.equals(NO_GROUP_BY)) {
318             neighborhoods.put("", factories);
319         } else {
320             throw new GroupByOperationNotSupportedException();
321         }
322         return neighborhoods;
323     }
