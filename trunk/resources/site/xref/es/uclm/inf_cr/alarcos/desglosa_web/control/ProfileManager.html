70      public static void addMappings(Metaclass metaclass, String jsonMappings) {
71       // Add mapping data to metaclass
72          List<Mapping> mappings = new ArrayList<Mapping>();
73          JSONArray mappingArray = (JSONArray) JSONSerializer.toJSON(jsonMappings);
74          for (int i = 0; i < mappingArray.size(); i++) {
75              JSONObject mappingObject = mappingArray.getJSONObject(i);
76              JSONObject entityAttribute = mappingObject.getJSONObject("entityAttribute");
77              Field column = new Field(entityAttribute.getString("type"), entityAttribute.getString("name"));
78              JSONObject modelAttribute = mappingObject.getJSONObject("modelAttribute");
79              Field attribute = new Field(modelAttribute.getString("type"), modelAttribute.getString("name"));
80              Object ratio = mappingObject.get("ratio");
81              if (ratio instanceof JSONNull) {
82                  ratio = null;
83              } else if (ratio instanceof Integer && attribute.getType().equals("float")) {
84                  ratio = ((Integer) ratio).floatValue();
85              } else if (ratio instanceof Double && attribute.getType().equals("float")) {
86                  ratio = ((Double) ratio).floatValue();
87              }
88              JSONArray jsonRules = mappingObject.getJSONArray("rules");
89              List<Rule> rules = new ArrayList<Rule>();
90              for (int j = 0; j < jsonRules.size(); j++) {
91                  JSONObject ruleJSONObject = jsonRules.getJSONObject(j);
92                  Object low = ruleJSONObject.get("low");
93                  if (low instanceof JSONNull) {
94                      low = null;
95                  } else if (low instanceof Integer && column.getType().equals("float")) {
96                      low = ((Integer) low).floatValue();
97                  } else if (low instanceof Double && column.getType().equals("float")) {
98                      low = ((Double) low).floatValue();
99                  }
100                 Object high = ruleJSONObject.get("high");
101                 if (high instanceof JSONNull) {
102                     high = null;
103                 } else if (high instanceof Integer && column.getType().equals("float")) {
104                     high = ((Integer) high).floatValue();
105                 } else if (high instanceof Double && column.getType().equals("float")) {
106                     high = ((Double) high).floatValue();
107                 }
108                 Object value = ruleJSONObject.get("value");
109                 if (value instanceof JSONNull) {
110                     value = null;
111                 } else if (value instanceof Integer && attribute.getType().equals("float_range")) {
112                     value = ((Integer) value).floatValue();
113                 } else if (value instanceof Double && attribute.getType().equals("float_range")) {
114                     value = ((Double) value).floatValue();
115                 }
116                 rules.add(new Rule(low, high, value));
117             }
118             Mapping mapping = new Mapping(column, attribute, null, ratio, rules);
119             mappings.add(mapping);
120         }
121         metaclass.setMappings(mappings);
122     }
